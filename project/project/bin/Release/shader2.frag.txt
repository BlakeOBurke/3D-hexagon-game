#version 330 core
out vec4 FragColor;  
in vec3 ourColor;
in float dist;
  
void main()
{
    float lightenFactor = 2;
    vec3 A = (ourColor+vec3(0.1,0,0)) * vec3(1,0,1);
    FragColor = vec4(min(A.x * lightenFactor, 1.0f), min(A.y * lightenFactor, 1.0f), min(A.z * lightenFactor, 1.0f), 1);
    //*vec3(0.5,0,0.65)
}
